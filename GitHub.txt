###  Generar ENTORNO VIRTUAL  ###

En la ubicación actual 
git init #Genera Env en ubicación
ls -al #muestra elementos ocultos

git add (nombre del archivo) #Agregar archivo al Repositorio
git rm --cached (nombre del archivo) #Eliminar de la lista al Repositorio
## DIFERENCIA ENTRE RM --CACHED Y RM ##

--CACHED Significa que está en memoria RAM, que está no guardado en la base de datos

git config #Muestra opciones de configuración
git config --list #Muestra configuraciones en lista
git config --list --show-origin #Muestra el origen del archivo de configuración
git config --global user.name "Brian Girod" #Configurar nombre
git config --global user.email "@gmail" #Configura email
